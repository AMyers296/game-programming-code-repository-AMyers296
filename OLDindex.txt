        <script>
                        
            // Engine Specific
            class Vector2 {
                constructor(x, y){
                    this.x = x
                    this.y = y
                }

                x
                y
            }



            
            // Engine Specific
            // Starting pos of Character

            let position = new Vector2()
            let position2 = new Vector2()

            function start(){
                //game Specifc
                position.x = 0
                position.y = 0

                position2.x = 100
                position2.y = 100
                
                gameLoop()
            }

            // Engine Specific
            function gameLoop(){
                update()
                draw()
                requestAnimationFrame(gameLoop)
            }

            //Engine Specific
            function update(){
                // Game specfic
                position.x++
                position.y++

                position2.x++
                position2.y++

            }

            function draw(){

                canvas.width = window.innerWidth
                canvas.height = window.innerHeight

                ctx.save()
                ctx.translate(position.x,position.y)

                ctx.beginPath()
                ctx.moveTo(50,60)
                ctx.lineTo(35,65)
                ctx.lineTo(50,35)
                ctx.lineTo(65,75)
                ctx.lineTo(50,60)

                ctx.lineWidth = 5
                ctx.strokeStyle = "purple"
                ctx.stroke()

                ctx.fillStyle = "yellow"
                ctx.fill()
                ctx.restore()

                ctx.save()
                ctx.translate(position2.x,position2.y)

                ctx.beginPath()
                ctx.moveTo(50,60)
                ctx.lineTo(35,65)
                ctx.lineTo(50,35)
                ctx.lineTo(65,75)
                ctx.lineTo(50,60)

                ctx.lineWidth = 5
                ctx.strokeStyle = "red"
                ctx.stroke()

                ctx.fillStyle = "orange"
                ctx.fill()
                ctx.restore()

            }

            // start()
            Engine.currentScene = new MainScene()
            Engine.start()

        </script>
